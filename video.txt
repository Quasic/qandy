               x = column              a = asci character

            pokeCell(x,y,a) — poke character(s) to a cell (faster)
          pokeChar(x,y,a,n) — poke character(s) to cells (slower)
  
            pokeFG(x,y,c,n) — set foreground color
            pokeFG(x,y,c,n) — set background color
     pokeColor(x,y,fg,bg,n) — set foreground and background color 
          pokeBold(x,y,s,n) — set bold attribute       
       pokeInverse(x,y,s,n) — set inverse attribute

              pokeCursor(t) — poke text at cursor's position          
            pokeText(x,y,t) — poke text at x,y (0,0 if not supplied)
  
                pokeInput() — poke input line at cursor position
              pokeSelect(s) — set inverse attribute for text selection
                  
           pokeRefresh(x,y) — update DOM cell, if x/y omited refresh entire DOM
      pokeRefreshRow(x,y,n) — update row of DOM cells 
             
            pokeAttr(x,y,h) — sets attibute to hex byte
         pokeAttrBit(x,y,c) — sets attibute bit



            pokeAttr(x,y,h) — sets attribute to hex mask
         pokeAttrBit(x,y,c) — sets attribute bit

                  peek(x,y) — return character at cell (x,y)
              peekAttr(x,y) — return numeric ATTR bitmask for cell (x,y)
           peekInverse(x,y) — return boolean whether inverse is active at cell
             peekColor(x,y) — return color object for cell
